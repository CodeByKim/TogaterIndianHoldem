115 void tcp_server::process_packet(const int n_session_id, const char * p_data)
116 {
117  packet_header *p_header = (packet_header *)p_data;
118  session *request_session = get_session(n_session_id);
119  switch (p_header->type)
120  {
121  case message_type::FRIENDS_REQ:
122  {
123  friends_manager_.process_friends_function(get_session(n_session_id), &p_data[packet_header_size], p_header->size);
124  break;
125  }
126  case message_type::PLAY_FRIENDS_REL:
127  {
128  match_manager_.process_matching_with_friends(request_session, &p_data[packet_header_size], p_header->size);
129  break;
130  }
131  case message_type::PLAY_RANK_REQ:
132  {
133  match_manager_.process_matching(request_session, &p_data[packet_header_size], p_header->size);
134  break;
135  }
136  case message_type::JOIN_REQ:
137  {
138  friends_manager_.lobby_login_process(request_session, &p_data[packet_header_size], p_header->size);
139  break;
140  }
141  case message_type::LOGOUT_REQ:
142  {
143  friends_manager_.lobby_logout_process(request_session, &p_data[packet_header_size], p_header->size);
144  break;
145  }
146  case message_type::MATCH_CONFIRM:
147  {
148  if (match_manager_.process_match_confirm(request_session, &p_data[packet_header_size], p_header->size))
149  {
150  close_session(n_session_id);
151  }
152  break;
153  }
154  case message_type::ERROR_MSG:
155  {
156  process_config(get_session(n_session_id), &p_data[packet_header_size], p_header->size);
157  break;
158  }
159  default:
160  break;
161  }
162 }
